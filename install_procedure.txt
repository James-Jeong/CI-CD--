9) Harbor 설치
(root 계정에 설치 권장)

wget https://github.com/goharbor/harbor/releases/download/v2.5.0/harbor-online-installer-v2.5.0.tgz
sudo cp harbor-online-installer-v2.5.0.tgz /var/lib
sudo tar -xzf harbor-online-installer-v2.5.0.tgz
cd harbor
sudo cp harbor.yml.tmpl harbor.yml
sudo vi harbor.yml
> hostname: [공인 IP]
> HTTPS 설정 : openssl
  
mkdir -p ~/certs
cd ~/certs
openssl genrsa -out ca.key 4096
openssl req -x509 -new -nodes -sha512 -days 3650 -subj "/C=KR/ST=Seoul/L=Seoul/O=Uangel/OU=Media/CN=[도메인 또는 IP]" -key ca.key -out ca.crt
openssl genrsa -out [도메인 또는 IP].key 4096
openssl req -sha512 -new -subj "/C=KR/ST=Seoul/L=Seoul/O=Uangel/OU=Media/CN=[도메인 또는 IP]" -key [도메인 또는 IP].key -out [도메인 또는 IP].csr

cat > v3_ext.cnf <<-EOF
authorityKeyIdentifier=keyid,issuer
basicConstraints=CA:FALSE
keyUsage = digitalSignature, nonRepudiation, keyEncipherment, dataEncipherment
extendedKeyUsage = serverAuth
subjectAltName = @alt_names

[alt_names]
<DNS.1=[도메인]> or <IP.1=[IP]>
EOF

openssl x509 -req -sha512 -days 3650 -extfile v3_ext.cnf -CA ca.crt -CAkey ca.key -CAcreateserial -in [도메인 또는 IP].csr -out [도메인 또는 IP].crt

# For docker
openssl x509 -inform PEM -in [도메인 또는 IP].crt -out [도메인 또는 IP]m.cert

sudo vi /usr/local/harbor/harbor.xml
> https 수정
- port : 1443
- certifacte : /etc/pki/ca-trust/source/anchors/[도메인 또는 IP].crt
- private_key : /etc/pki/ca-trust/source/anchors/[도메인 또는 IP].key

sudo ./install.sh

@ 변경내역 반영
sudo ./prepare 
> docker-compose 재실행 
sudo docker-compose down -v 
sudo docker-compose up -d
